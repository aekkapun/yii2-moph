SELECT year(t.DATETIME_DISCH) as yy, 
month(t.DATETIME_DISCH) as mm, 
count(t.AN) as cnt, 
sum(t.ADJRW) as sumadj, 
avg(t.ADJRW) as cmi , 
sum(t.ACTLOS) as los, 
avg(t.ACTLOS) as los_per_case
FROM admission t  
GROUP BY yy,mm
